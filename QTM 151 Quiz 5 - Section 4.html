    the date column and freq to "m" (month)
  • Follow the instructions in Lecture 21 to compute an aggregate dataset

# Write your own code
month_config = pd.Grouper(key = 'date',freq = 'm')

monthlydata = (bitcoin
              .groupby(month_config)
              .agg(cb_to_usd_std = ("cb_to_usd","std"))
              .reset_index())
display(monthlydata)
plt.plot("date","cb_to_usd_std",
         data = monthlydata.sort_values("date",ascending=True))
plt.title("Standard deviation of CoinBase bitcoin over Month")
